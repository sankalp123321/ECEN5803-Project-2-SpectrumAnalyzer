177 pFunctionPointer_t add_common(T *tptr, void (T::*mptr)(void), IRQn_Type irq, bool front = false)
178 {
179 _mutex.lock();
180 int irq_pos = get_irq_index(irq);
181 bool change = must_replace_vector(irq);
182
183 pFunctionPointer_t pf = front ? _chains[irq_pos]->add_front(tptr, mptr) : _chains[irq_pos]->add(tptr, mptr);
184 if (change) {
185 NVIC_SetVector(irq, (uint32_t)&InterruptManager::static_irq_helper);
186 }
187 _mutex.unlock();
188 return pf;
189 }
