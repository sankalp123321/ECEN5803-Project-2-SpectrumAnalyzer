564 static __INLINE uint32_t arm_recip_q31(
565 q31_t in,
566 q31_t * dst,
567 q31_t * pRecipTable)
568 {
569 q31_t out;
570 uint32_t tempVal;
571 uint32_t index, i;
572 uint32_t signBits;
573
574 if(in > 0)
575 {
576 signBits = ((uint32_t) (__CLZ( in) - 1));
577 }
578 else
579 {
580 signBits = ((uint32_t) (__CLZ(-in) - 1));
581 }
582
583 /* Convert input sample to 1.31 format */
584 in = (in << signBits);
585
586 /* calculation of index for initial approximated Val */
587 index = (uint32_t)(in >> 24);
588 index = (index & INDEX_MASK);
589
590 /* 1.31 with exp 1 */
591 out = pRecipTable[index];
592
593 /* calculation of reciprocal value */
594 /* running approximation for two iterations */
595 for (i = 0u; i < 2u; i++)
596 {
597 tempVal = (uint32_t) (((q63_t) in * out) >> 31);
598 tempVal = 0x7FFFFFFFu - tempVal;
599 /* 1.31 with exp 1 */
600 /* out = (q31_t) (((q63_t) out * tempVal) >> 30); */
601 out = clip_q63_to_q31(((q63_t) out * tempVal) >> 30);
602 }
603
604 /* write output */
605 *dst = out;
606
607 /* return num of signbits of out = 1/in value */
608 return (signBits + 1u);
609 }

615 static __INLINE uint32_t arm_recip_q15(
616 q15_t in,
617 q15_t * dst,
618 q15_t * pRecipTable)
619 {
620 q15_t out = 0;
621 uint32_t tempVal = 0;
622 uint32_t index = 0, i = 0;
623 uint32_t signBits = 0;
624
625 if(in > 0)
626 {
627 signBits = ((uint32_t)(__CLZ( in) - 17));
628 }
629 else
630 {
631 signBits = ((uint32_t)(__CLZ(-in) - 17));
632 }
633
634 /* Convert input sample to 1.15 format */
635 in = (in << signBits);
636
637 /* calculation of index for initial approximated Val */
638 index = (uint32_t)(in >> 8);
639 index = (index & INDEX_MASK);
640
641 /* 1.15 with exp 1 */
642 out = pRecipTable[index];
643
644 /* calculation of reciprocal value */
645 /* running approximation for two iterations */
646 for (i = 0u; i < 2u; i++)
647 {
648 tempVal = (uint32_t) (((q31_t) in * out) >> 15);
649 tempVal = 0x7FFFu - tempVal;
650 /* 1.15 with exp 1 */
651 out = (q15_t) (((q31_t) out * tempVal) >> 14);
652 /* out = clip_q31_to_q15(((q31_t) out * tempVal) >> 14); */
653 }
654
655 /* write output */
656 *dst = out;
657
658 /* return num of signbits of out = 1/in value */
659 return (signBits + 1);
660 }
