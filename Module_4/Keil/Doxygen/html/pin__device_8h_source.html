72__STATIC_INLINE void LL_GPIO_SetAFPin_0_7(GPIO_TypeDef *GPIOx, uint32_t Pin, uint32_t Alternate)
73{
74 MODIFY_REG(GPIOx->AFR[0], (0xFU << (POSITION_VAL(Pin) * 4U)),

76}
77__STATIC_INLINE void LL_GPIO_SetAFPin_8_15(GPIO_TypeDef *GPIOx, uint32_t Pin, uint32_t Alternate)

82__STATIC_INLINE void LL_GPIO_SetPinMode(GPIO_TypeDef *GPIOx, uint32_t Pin, uint32_t Mode)
83{
84 MODIFY_REG(GPIOx->MODER, (GPIO_MODER_MODER0 << (POSITION_VAL(Pin) * 2U)), (Mode << (POSITION_VAL(Pin) * 2U)));
85}
86__STATIC_INLINE uint32_t LL_GPIO_GetPinMode(GPIO_TypeDef *GPIOx, uint32_t Pin)
87{

89}

113static inline void stm_pin_PullConfig(GPIO_TypeDef *gpio, uint32_t ll_pin, uint32_t pull_config)
114{
115 switch (pull_config) {
116 case GPIO_PULLUP:
117 LL_GPIO_SetPinPull(gpio, ll_pin, LL_GPIO_PULL_UP);
118 break;
119 case GPIO_PULLDOWN:
120 LL_GPIO_SetPinPull(gpio, ll_pin, LL_GPIO_PULL_DOWN);
121 break;
122 default:
123 LL_GPIO_SetPinPull(gpio, ll_pin, LL_GPIO_PULL_NO);
124 break;
125 }
126}

128static inline void stm_pin_SetAFPin(GPIO_TypeDef *gpio, PinName pin, uint32_t afnum)
129{
130 uint32_t ll_pin = ll_pin_defines[STM_PIN(pin)];
131
132 if (STM_PIN(pin) > 7) {
133 LL_GPIO_SetAFPin_8_15(gpio, ll_pin, afnum);
134 } else {
135 LL_GPIO_SetAFPin_0_7(gpio, ll_pin, afnum);
136 }
137}
