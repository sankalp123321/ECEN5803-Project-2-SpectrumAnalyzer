579 CMSIS_INLINE __STATIC_INLINE uint32_t arm_recip_q31(
580 q31_t in,
581 q31_t * dst,
582 q31_t * pRecipTable)
583 {
584 q31_t out;
585 uint32_t tempVal;
586 uint32_t index, i;
587 uint32_t signBits;
588
589 if (in > 0)
590 {
591 signBits = ((uint32_t) (__CLZ( in) - 1));
592 }
593 else
594 {
595 signBits = ((uint32_t) (__CLZ(-in) - 1));
596 }
597
598 /* Convert input sample to 1.31 format */
599 in = (in << signBits);
600
601 /* calculation of index for initial approximated Val */
602 index = (uint32_t)(in >> 24);
603 index = (index & INDEX_MASK);
604
605 /* 1.31 with exp 1 */
606 out = pRecipTable[index];
607
608 /* calculation of reciprocal value */
609 /* running approximation for two iterations */
610 for (i = 0u; i < 2u; i++)
611 {
612 tempVal = (uint32_t) (((q63_t) in * out) >> 31);
613 tempVal = 0x7FFFFFFFu - tempVal;
614 /* 1.31 with exp 1 */
615 /* out = (q31_t) (((q63_t) out * tempVal) >> 30); */
616 out = clip_q63_to_q31(((q63_t) out * tempVal) >> 30);
617 }
618
619 /* write output */
620 *dst = out;
621
622 /* return num of signbits of out = 1/in value */
623 return (signBits + 1u);
624 }

631 q15_t in,
632 q15_t * dst,
633 q15_t * pRecipTable)
634 {
635 q15_t out = 0;
636 uint32_t tempVal = 0;
637 uint32_t index = 0, i = 0;
638 uint32_t signBits = 0;
639
640 if (in > 0)
641 {
642 signBits = ((uint32_t)(__CLZ( in) - 17));
643 }
644 else
645 {
646 signBits = ((uint32_t)(__CLZ(-in) - 17));
647 }
648
649 /* Convert input sample to 1.15 format */
650 in = (in << signBits);
651
652 /* calculation of index for initial approximated Val */
653 index = (uint32_t)(in >> 8);
654 index = (index & INDEX_MASK);
655
656 /* 1.15 with exp 1 */
657 out = pRecipTable[index];
658
659 /* calculation of reciprocal value */
660 /* running approximation for two iterations */
661 for (i = 0u; i < 2u; i++)
662 {
663 tempVal = (uint32_t) (((q31_t) in * out) >> 15);
664 tempVal = 0x7FFFu - tempVal;

667 /* out = clip_q31_to_q15(((q31_t) out * tempVal) >> 14); */
668 }
669
670 /* write output */
671 *dst = out;
672
673 /* return num of signbits of out = 1/in value */
674 return (signBits + 1);
675 }

6794 f01 = pData[index + 1];

6802 f11 = pData[index + 1];
6803

6806 b2 = f01 - f00;
